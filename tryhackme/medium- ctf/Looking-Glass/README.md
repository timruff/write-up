import os

debut = 9000
fin = 14000
milieu = 0


while True :
    milieu = int((fin-debut)/2+debut)
    cmd = "ssh lookingglass.thm -o StrictHostKeyChecking=no  -p " + str(milieu)
    resultat = os.popen(cmd).read()
    
    if resultat.find('Lower') >= 0:
        debut = milieu
    elif resultat.find('Higher') >= 0:
        fin = milieu
    else:
        print(milieu)
        print(resultat)
        break
